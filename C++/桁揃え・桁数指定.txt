#include<stdio.h>
int main(void) {
	printf("%d\n", 1);
	printf("%d\n", 10);
	printf("%d\n", 100);

	printf("\n");

	printf("%4d\n", 0);		//%とdの間に表示桁数が指定された場合は、表示文字数内に右詰めで表示できる。
	printf("%4d\n", 00);
	printf("%4d\n", 000);

	printf("\n");

	printf("%lf\n", 12.3);		//表示桁の指定がない場合は、小数は下6桁まで表示される。
	printf("%.4lf\n", 12.3);	//%とlfの間に.4のように記述すると小数点以下の文字数を指定できる。
	printf("%6.2lf\n", 12.3);	//%とlfの間に6.2のように記述すると表示エリアは6文字分、小数点以下は2桁になる。小数点も表示桁数に含まれるので注意。

	printf("\n");

	printf("%2d\n", 123456);	//データが表示文字数を超えてしまった場合、文字数の指定は無視される。
	return 0;
}
/*↓実行結果↓
1
10
100

      0
    00
  000

12.300000
12.3000
  12.30

123456
*/
